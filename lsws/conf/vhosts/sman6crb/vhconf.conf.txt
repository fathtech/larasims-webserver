enablegzip 1
docroot $VH_ROOT

accesscontrol  {
    allow *
}

extprocessor sman6crb-app {
    retrytimeout 7200
    respbuffer 0
    inittimeout 60
    address sman6crb_app-server_1
    maxconns 1500
    type proxy
}

index  {
    autoindex 0
    autoindexuri /_autoindex/default.php
    useserver 0
    indexfiles index.html, index.php
}

expires  {
    enableexpires 1
}

rewrite  {
    enable 1
    autoloadhtaccess 1
    loglevel 0
}

accesslog $SERVER_ROOT/logs/$VH_NAME.access.log {
    useserver 0
    rollingsize 100M
    compressarchive 1
    keepdays 7
}

context / {
    allowbrowse 1
    location $DOC_ROOT/

    rewrite  {
        unknownkeywords rewritefile .htaccess
        rules rewritefile .htaccess
    }
}

errorlog  {
    useserver 1
}

vhssl  {
    keyfile /root/.acme.sh/certs/$VH_NAME/$VH_NAME.key
    certfile /root/.acme.sh/certs/$VH_NAME/fullchain.cer
    certchain 1
}
