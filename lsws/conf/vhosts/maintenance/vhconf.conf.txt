vhdomain *
docroot $VH_ROOT
enablegzip 1

accesscontrol  {
    allow *
}

vhssl  {
    certfile /root/.acme.sh/certs/$VH_NAME/fullchain.cer
    certchain 1
    keyfile /root/.acme.sh/certs/$VH_NAME/$VH_NAME.key
}

rewrite  {
    autoloadhtaccess 1
    enable 1
    loglevel 0
}

index  {
    autoindexuri /_autoindex/default.php
    useserver 0
    autoindex 0
    indexfiles index.html, index.php
}

accesslog $SERVER_ROOT/logs/$VH_NAME.access.log {
    rollingsize 100M
    useserver 0
    compressarchive 1
    keepdays 7
}

context / {
    allowbrowse 1
    location $DOC_ROOT/

    rewrite  {
        rules rewritefile .htaccess
        unknownkeywords rewritefile .htaccess
    }
}

expires  {
    enableexpires 1
}

errorlog  {
    useserver 1
}
